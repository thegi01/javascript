Array.prototype.forEach()

The forEach() method executes a provided function once for each array element.

		const arr = ['a', 'b', 'c'];

		arr.forEach(function(element) {
		    console.log(element);
		});

		// a
		// b
		// c

Syntax

		arr.forEach(function callback(currentValue[, index[, array]]) {
		    //your iterator
		}[, thisArg]);
